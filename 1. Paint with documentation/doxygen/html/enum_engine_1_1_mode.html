Point2D zeroVector = new Point2D(0, 1);
double angle = zeroVector.angle(mouse.subtract(new Point2D(tShape.getLayoutX(), tShape.getLayoutY())));
if(mouse.getX() > tShape.getLayoutX())
angle = 360 - angle;
tShape.setRotate(angle);
}
